import asyncio
from app.services.generator import code_generation_service

async def generate_code(prompt: str):
    """Generate code using AI models"""
    result = await code_generation_service.generate(
        prompt=prompt,
        language="python",
        model="mistral-ai/Codestral-2501"
    )
    
    # Validate syntax and quality
    validation = await code_generation_service.validate(
        code=result.code,
        language="python"
    )
    
    if validation.is_valid:
        return result.code
    else:
        # Fix any detected issues
        return await code_generation_service.improve(
            code=result.code,
            issues=validation.errors
        )

# Example prompt
result = asyncio.run(generate_code(
    "Create a function to calculate fibonacci numbers"
))
print(result)
